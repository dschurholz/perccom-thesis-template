\newcommand{\keywordList}[2]{
    \newcount\idx
    \newcount\lastEl
    \idx=0
    \lastEl=#1
    \advance \lastEl-1
    \loop
        \keywords{\the\idx}\ifnum \idx<\lastEl#2\fi
    \advance \idx+1
    \ifnum \idx<#1
    \repeat
}
